MEX_worsen_pandemic = {
	if = {
		limit = {
			has_idea = MEX_covid_initial
		}
		swap_ideas = {
			remove_idea = MEX_covid_initial
			add_idea = MEX_covid_uncontrolled
		}
	}
	else_if = {
		limit = { 
			has_idea = MEX_covid_uncontrolled
		}
		swap_ideas = {
			remove_idea = MEX_covid_uncontrolled
			add_idea = MEX_covid_massive
		}
	}
	else_if = {
		limit = { 
			has_idea = MEX_covid_massive
		}
		swap_ideas = {
			remove_idea = MEX_covid_massive
			add_idea = MEX_covid_unstoppable
		}
	}
}
